  components: {
    // components
  },
  setup(props, { emit }) {
    // 데이터를 자동으로 반응형으로 만들어 줌
    const data = reactive({
      count: 0,
    });
 
    /* props.user 값을 반응형으로 가져올 수 있음
     * getUserRepositories를 실행하면 repositories에 최신 값이 들어가고
     * ref에 따라서 화면이 update 됨 */
    const repositories = ref([])
      const getUserRepositories = async () => {
        repositories.value = await fetchUserRepositories(props.user)
      }
 
    const incrementCount = () => data.count++;
    const onKeydownPlus = (e) => e?.keyCode === 187 && incrementCount();
    const multiplyCount = computed(() => count.value * 2);
 
    watch(count, (val) => console.log("count", val));
 
    onMounted(() => {
      document.addEventListener("keydown", onKeydownPlus);
    });
    onUnmounted(() => {
      document.removeEventListener("keydown", onKeydownPlus);
    });
    // onBefore###, on###ed
    // Mount, Update, Unmount, (Activated, Deactivated, ErrorCaptured)
 
    return {
      ...toRefs(data),
      multiplyCount,
      repositories,
      getUserRepositories
    };
  },
};
</script>
